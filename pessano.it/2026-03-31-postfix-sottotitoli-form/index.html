Fix attributi: dettaglio tecnico

// Il separatore usato dal cliente per le misure era "I" (lettera I maiuscola) // WooCommerce di default usa "|" (pipe) — il codice originale cercava solo pipe // Dato in database: "9 I 12 I 15" // Prima del fix: array(1) { [0] => "9 I 12 I 15" } → count=1 → niente dropdown // Dopo il fix: array(3) { "9", "12", "15" } → count=3 → dropdown generato private static function parseAttributeValues(array $rawValues): array { if (count($rawValues) === 1) { $val = $rawValues[0]; if (strpos($val, '|') !== false) { return array_map('trim', explode('|', $val)); } if (preg_match('/\bI\b/', $val)) { return array_map('trim', preg_split('/\s+I\s+/', $val)); } } return array_map('trim', $rawValues); }
